function transition!(x, rng, k, x_prev, scratch)
    @unpack r, K, Ïƒ_p, xâ‚€ = scratch.par

    Îµ_t = rand(rng, Normal(0, Ïƒ_p))

function log_transition(k, x_prev, x, scratch)
    @unpack r, K, Ïƒ_p, xâ‚€ = scratch.par
    
    old_x = k == 1 ? xâ‚€ : x_prev.s
    Îµ = x.s / old_x - 1 - r*(1 - old_x/K)

function log_potential(k, x, scratch)
    if x.s <= 0
        return -Inf
    end

    Î± = x.s^2 / scratch.par.Ïƒ_o^2
    Î¸ = scratch.par.Ïƒ_o^2 / x.s
    logpdf(Gamma(Î±, Î¸), scratch.y[k])
end

function set_param!(scratch, Î¸)

function post_pred(chn, ts; process_noise = false)
    p = get(chn; section=:parameters)
    nsamples = length(p[1])

    T = length(ts)
    X = Array{Float64}(undef, T, nsamples)

    for i in 1:nsamples
        x = Array{Float64}(undef, T)
        r = p.r[i]
        K = p.K[i]
        Ïƒ_p = p.Ïƒ_p[i]
        Ïƒ_o = p.Ïƒ_o[i]
        xâ‚€ = p.xâ‚€[i]

        Îµ = zeros(T)
        if process_noise
            Îµ = rand(Normal(0, Ïƒ_p), T)
        end

        for t in ts
            x_lastt = t == 1 ? xâ‚€ : x[t-1]
            x[t] = (1 + r*(1 - x_lastt/K) + Îµ[t]) * x_lastt
        end
        X[:, i] = x
    end

    mapslices(x_t -> quantile(x_t, [0.025, 0.25, 0.5, 0.75, 0.975]), X, dims = 2)
end

function sample_hiddenstate(data, n)
    T, nsamples, nchains = size(data)
    ntotalsamples = nsamples * nchains
    d = deepcopy(data)
    d = reshape(d, (T, ntotalsamples))
    d[:, sample(1:ntotalsamples, n; replace = false)]
end

function log_transition(k, x_prev, x, scratch)
    @unpack r, K, Ïƒ_p, xâ‚€ = scratch.par
    
    old_x = k == 1 ? xâ‚€ : x_prev.s
    Îµ = x.s / old_x - 1 - r*(1 - old_x/K)
    
    return logpdf(Normal(0, Ïƒ_p), Îµ)
end

function log_potential(k, x, scratch)
    if x.s <= 0
        return -Inf
    end

    Î± = x.s^2 / scratch.par.Ïƒ_o^2
    Î¸ = scratch.par.Ïƒ_o^2 / x.s
    logpdf(Gamma(Î±, Î¸), scratch.y[k])
end

function set_param!(scratch, Î¸)
    scratch.par.r = exp(Î¸.log_r)
    scratch.par.K = exp(Î¸.log_K)
    scratch.par.Ïƒ_p = exp(Î¸.log_sigma_p)
    scratch.par.Ïƒ_o = exp(Î¸.log_sigma_o)
    scratch.par.xâ‚€ = exp(Î¸.log_xâ‚€)
end

function prior(theta)
    (logpdf(Normal(log(0.1), 1.0), theta.log_r) +
     logpdf(Normal(log(200.0), 0.5), theta.log_K) +
     logpdf(Normal(log(0.1), 0.5), theta.log_sigma_p) +
     logpdf(Normal(log(10.0), 1.0), theta.log_sigma_o) +
     logpdf(Normal(log(10.0), 0.5), theta.log_xâ‚€))
end

function sample_prior()
    LVector(log_r = rand(Normal(log(0.1), 1.0)),
            log_K = rand(Normal(log(200.0), 0.5)),
            log_sigma_p = rand(Normal(log(0.1), 0.5)),
            log_sigma_o = rand(Normal(log(10.0), 1.0)),
            log_xâ‚€ = rand(Normal(log(10.0), 0.5)))
end

function sample_initial_values(sol)
    f() = randn() * 0.01 + 1
    p = sol.parameter
    
    LVector(log_r = log(p[:r] * f()),
        log_K = log(p[:K] * f()),
        log_sigma_p = log(p[:Ïƒ_p] * f()),
        log_sigma_o = log(p[:Ïƒ_o] * f()),
        log_xâ‚€ = log(p[:xâ‚€] * f()))
end

function sample_prior_contrain(n)
    (; r = exp.(rand(Normal(log(0.1), 1.0), n)),
       K = exp.(rand(Normal(log(200.0), 0.5), n)),
       Ïƒ_p = exp.(rand(Normal(log(0.1), 0.5), n)),
       Ïƒ_o = exp.(rand(Normal(log(10.0), 1.0), n)),
       xâ‚€ = exp.(rand(Normal(log(10.0), 0.5), n)))
end

function post_pred(chn, ts; process_noise = false)
    p = get(chn; section=:parameters)
    nsamples = length(p[1])

    T = length(ts)
    X = Array{Float64}(undef, T, nsamples)

    for i in 1:nsamples
        x = Array{Float64}(undef, T)
        r = p.r[i]
        K = p.K[i]
        Ïƒ_p = p.Ïƒ_p[i]
        Ïƒ_o = p.Ïƒ_o[i]
        xâ‚€ = p.xâ‚€[i]

        Îµ = zeros(T)
        if process_noise
            Îµ = rand(Normal(0, Ïƒ_p), T)
        end

        for t in ts
            x_lastt = t == 1 ? xâ‚€ : x[t-1]
            x[t] = (1 + r*(1 - x_lastt/K) + Îµ[t]) * x_lastt
        end
        X[:, i] = x
    end

    mapslices(x_t -> quantile(x_t, [0.025, 0.25, 0.5, 0.75, 0.975]), X, dims = 2)
end

function sample_hiddenstate(data, n)
    T, nsamples, nchains = size(data)
    ntotalsamples = nsamples * nchains
    d = deepcopy(data)
    d = reshape(d, (T, ntotalsamples))
    d[:, sample(1:ntotalsamples, n; replace = false)]
end
