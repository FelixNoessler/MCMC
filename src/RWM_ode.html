Metropolis-Algorithm for parameters estimation of ODEs

the script is adapted from the Bayesian Estimation of Differential Equations tutorial from Turing.jl, but instead of relying on the Nuts algorithm of Turing.jl, a simple Metroplis algorithm is coded here from scratch

function lotka_volterra(du, u, p, t)
    Î±, Î², Î³, Î´ = p
    x, y = u
    
    du[1] = (Î± - Î² * y) * x 
    du[2] = (Î´ * x - Î³) * y 

    return nothing
end

function generate_data(rng; p)
    u0 = [1.0, 1.0]
    tspan = (0.0, 10.0)
    prob = ODEProblem(lotka_volterra, u0, tspan, p)
    
    sol = solve(prob, Tsit5();
                saveat=0.1)

    return estim_dat, sol.t
end

4 Function to calculate the unnormalized posterior density

function unnormalized_posterior(Î¸, prior_dists, data, t)
    Ïƒ, Î±, Î², Î³, Î´ = Î¸
    nparameter = length(Î¸)
    
    ## prior 
    if Ïƒ <= 0
        return -Inf
    end
    
    prior = 0
    for i in 1:nparameter
        prior += logpdf(prior_dists[i], Î¸[i])
    end
    if prior == -Inf
        return -Inf
    end

    ## likelihood
    p = [Î±, Î², Î³, Î´]
    u0 = [1.0, 1.0]
    tspan = (0.0, 10.0)
    prob = ODEProblem(lotka_volterra, u0, tspan, p)
    predicted = solve(prob, Tsit5(); p=p, saveat=t)

    likelihood = 0
    for i in 1:length(predicted)
        likelihood += logpdf(MvNormal(predicted[i], Ïƒ^2 * I), data[:, i])
    end
    
    return prior + likelihood
end

5 Function to simulate the Markov chains

function run_chains(rng, data, t; 
                    Ïƒ_prop,
                    nchains=5,
                    nsamples=5_000)

    ## priors
    Ïƒ_prior = truncated(InverseGamma(2, 3); lower=0, upper=1)
    Î±_prior = truncated(Normal(1.5, 0.5); lower=0.8, upper=2.5)
    Î²_prior = truncated(Normal(1.2, 0.5); lower=0, upper=2)
    Î³_prior = truncated(Normal(3.0, 0.5); lower=1, upper=4)
    Î´_prior = truncated(Normal(1.0, 0.5); lower=0, upper=2)
    prior_dists = [Ïƒ_prior, Î±_prior, Î²_prior, Î³_prior, Î´_prior]
    
    nparameter = 5
    accepted_Î¸ = zeros(nchains, nparameter, nsamples)
    accepted = zeros(nchains)
    Î¸ = zeros(nchains, nparameter)
    
    Threads.@threads for n in 1:nchains        
        ## start values for the parameters in the chain
        ## rough guesses are used here 
        ## it would also possible to use the prior distributions as follows: 
        ## for i in 1:nparameter
        ##     Î¸[n, i] = rand(rng, prior_dists[i])
        ## end
        Î¸[n, :] = [0.7, 1.4, 0.9, 3.1, 1.1] .+ rand(rng, Normal(0, 0.1), 5)
        post = unnormalized_posterior(Î¸[n, :], prior_dists, data, t)
        
        for k in 1:nsamples
            ## new proposal
            proposal_dist = MvNormal(Î¸[n, :], Ïƒ_prop)
            Î¸star = rand(rng, proposal_dist) 
            
            ## evaluate prior + likelihood
            poststar = unnormalized_posterior(Î¸star, prior_dists, data, t)
            
            ## Metropolis ratio
            ratio = poststar - post

            if log(rand(rng)) < min(ratio, 1)
                accepted[n] += 1
                Î¸[n, :] = Î¸star
                post = poststar
            end
            
            accepted_Î¸[n, :, k] = Î¸[n, :]
        end
 
    end
    
    return accepted_Î¸, accepted / nsamples
end

function plot_trace_dens(; Î¸, burnin=nothing)
    fig = Figure(size=(800, 800))
    
    titles = ["Ïƒ", "Î±", "Î²", "Î³", "Î´"]
    nchains, nparameter, nsamples = size(Î¸)
    burnin = isnothing(burnin) ? max(Int(0.5*nsamples), 1) : burnin
    
    for i in 1:nparameter
        Axis(fig[i,1]; title = titles[i])
        
        for n in 1:nchains
            lines!((burnin:nsamples) .- burnin, Î¸[n, i, burnin:end];
                color=(Makie.wong_colors()[n], 0.5))
        end
        
        Axis(fig[i,2])
        for n in 1:nchains
            density!(Î¸[n, i, burnin:end];
                    bins=20, 
                    color= (Makie.wong_colors()[n], 0.05),
                    strokecolor = (Makie.wong_colors()[n], 1),
                    strokewidth = 2, strokearound = false)
        end

    end
    rowgap!(fig.layout, 1, 5)
    
    return fig
end

function posterior_check(rng; Î¸, data, t, p, npost_samples=500, burnin=nothing)

    nchains, nparameter, nsamples = size(Î¸)
    burnin = isnothing(burnin) ? max(Int(0.5*nsamples), 1) : burnin
    
    u0 = [1.0, 1.0]
    tspan = (0.0, 10.0)
    
    fig = Figure()
    Axis(fig[1,1]; xlabel="Time", ylabel="Density")
    
    ## select posterior draws and plot solutions
    selected_chains = rand(rng, 1:nchains, npost_samples)
    selected_samples = rand(rng, burnin:nsamples, npost_samples)
    for k in 1:npost_samples
        Î¸i = Î¸[selected_chains[k], :, selected_samples[k]]
        p_i = Î¸i[2:5]
 
        prob = ODEProblem(lotka_volterra, u0, tspan, p_i)
        sol = solve(prob, Tsit5(); saveat=0.01)
        
        lines!(sol.t, sol[1, :], color=(Makie.wong_colors()[1], 0.05))
        lines!(sol.t, sol[2, :], color=(Makie.wong_colors()[2], 0.05))
    end
    
    ## true solution
    prob = ODEProblem(lotka_volterra, u0, tspan, p)
    sol = solve(prob, Tsit5(); p=p, saveat=0.01)
    
    lines!(sol.t, sol[1, :], 
           color=:black,
           linewidth=2)
    lines!(sol.t, sol[2, :], 
           color=:black,
           linewidth=2)
    
    ## measured data
    scatter!(t, data[1, :])
    scatter!(t, data[2, :])

   return fig
end
